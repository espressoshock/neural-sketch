\begin{nskFigure}[]

			\begin{nskSwitch}[type=string]{\c}
				\nskCase[skip]{
					\nskBlock[
						last-pos-s={right=.0cm},
						border-type=dashed,
						shadow=false,
					]
				}
				\nskCase[dots]{
					\nskBlock[
						last-pos-s={right=0cm},
						text-center=$\cdots$,
						border-type=none, fill=none,
						shadow=false,
					]
				}
				\nskDefault{
					\nskBlock[
						last-pos-s={right=.5cm},
						text-north=\c,
						fill=nsk\c,
					]
				}
